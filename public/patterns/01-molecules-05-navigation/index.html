<script>

$(function() {

  var theTabs = $('.tab-link');
  var theTabSubMenus = $('.tab-submenu-list');
  var theTabSubMenuLinks = $('.tab-submenu-link');
  var theTabSubMenuContent = $('.tab-submenu-content');

  $(theTabs).click(function(){
    var theClickedTab = $(this);
    var theClickedTabId = theClickedTab.attr('data-tab');  //.attr('data-tab');
    selectTab(theClickedTab);
    showSubNav(theClickedTabId);
  });

  $(theTabSubMenuLinks).click(function() {
    var theClickedTabSubMenuLink = $(this);
    var theClickedTabSubMenuLinkId = theClickedTabSubMenuLink.attr('data-tab');
    selectTabSubMenuLink(theClickedTabSubMenuLink);
    showTabSubMenuContent(theClickedTabSubMenuLinkId);

  });

  function selectTab(tab){            
    $(theTabs).removeClass('active');
    $(tab).addClass('active');
  }

  function showSubNav(tab) {
    $(theTabSubMenus).removeClass('active');
    $('#' + tab).addClass('active');
  }

  function selectTabSubMenuLink(tab) {
    $(theTabSubMenuLinks).removeClass('active');
    $(tab).addClass('active');
  }

  function showTabSubMenuContent(tab) {
    $(theTabSubMenuContent).removeClass('active');
    $('#' + tab).addClass('active');
  }



}); // end function
</script>
